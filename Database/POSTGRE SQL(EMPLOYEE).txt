-- CREATING TABLES
CREATE TABLE DEPT(DEPT_ID INT,DNAME CHAR(10));

CREATE TABLE EMPLOYEE (EMPLOYEE_ID INT,FIRST_NAME CHAR(10) NOT NULL,LAST_NAME CHAR(10) NOT NULL,DOB DATE,DEPT_ID INT);

CREATE TABLE PHONE_TYPE (PHONE_TYPE_ID INT,PHONE_TYPE CHAR(10));

CREATE TABLE PHONE(PHONE_ID INT,EMPLOYEE_ID INT,PHONE_TYPE_ID INT,PHONE_NUMBER INT);

CREATE TABLE EMAIL_TYPE (EMAIL_TYPE_ID INT,EMAIL_TYPE CHAR(10));

CREATE TABLE EMAIL (EMAIL_ID INT,EMPLOYEE_ID INT,EMAIL_TYPE_ID INT,EMAIL CHAR(20));

-- APPLYING CONSTRAINTS

ALTER TABLE DEPT ADD CONSTRAINT PK_DEPT PRIMARY KEY(DEPT_ID);

ALTER TABLE EMPLOYEE ADD CONSTRAINT PK_EMPLOYEE PRIMARY KEY(EMPLOYEE_ID);

ALTER TABLE EMPLOYEE ADD CONSTRAINT FK_DEPT_ID FOREIGN KEY(DEPT_ID) REFERENCES DEPT(DEPT_ID);

ALTER TABLE PHONE_TYPE ADD CONSTRAINT PK_PHONE_TYPE PRIMARY KEY(PHONE_TYPE_ID);

ALTER TABLE PHONE ADD CONSTRAINT PK_PHONE PRIMARY KEY(PHONE_ID);

ALTER TABLE PHONE ADD CONSTRAINT FK_PH_EMPLOYEE_ID FOREIGN KEY(EMPLOYEE_ID) REFERENCES EMPLOYEE(EMPLOYEE_ID);

ALTER TABLE PHONE ADD CONSTRAINT FK_PHONE_TYPE_ID FOREIGN KEY(PHONE_TYPE_ID) REFERENCES PHONE_TYPE(PHONE_TYPE_ID);

ALTER TABLE EMAIL_TYPE ADD CONSTRAINT PK_EMAIL_TYPE PRIMARY KEY(EMAIL_TYPE_ID);

ALTER TABLE EMAIL ADD CONSTRAINT PK_EMAIL PRIMARY KEY(EMAIL_ID);

ALTER TABLE EMAIL ADD CONSTRAINT FK_EMAIL_EMPLOYEE_ID FOREIGN KEY(EMPLOYEE_ID) REFERENCES EMPLOYEE(EMPLOYEE_ID);

ALTER TABLE EMAIL ADD CONSTRAINT FK_EMAIL_TYPE_ID FOREIGN KEY(EMAIL_TYPE_ID) REFERENCES EMAIL_TYPE(EMAIL_TYPE_ID);

ALTER TABLE PHONE ADD PHONE_NUMBER CHAR(15);

-- SEQUENCES

CREATE SEQUENCE SQ_EMP START WITH 100 INCREMENT BY 1;

CREATE SEQUENCE SQ_DEPT START WITH 10 INCREMENT BY 10;

CREATE SEQUENCE SQ_PHONE START WITH 001 INCREMENT BY 1;

CREATE SEQUENCE SQ_EMAIL START WITH 111 INCREMENT BY 111;
